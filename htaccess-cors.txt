# BEGIN CORS for Capacitor iOS/Android App
<IfModule mod_headers.c>
    SetEnvIf Origin "^(https?://localhost(:[0-9]+)?|capacitor://localhost|ionic://localhost|https://tejcoms\.com)$" CORS_ORIGIN=$0
    Header always set Access-Control-Allow-Origin "%{CORS_ORIGIN}e" env=CORS_ORIGIN
    Header always set Access-Control-Allow-Methods "GET, POST, OPTIONS, PUT, DELETE"
    Header always set Access-Control-Allow-Headers "Authorization, Content-Type, X-WP-Nonce, X-Requested-With"
    Header always set Access-Control-Allow-Credentials "true"
    Header always set Access-Control-Expose-Headers "X-WP-Total, X-WP-TotalPages, Link"
    Header merge Vary "Origin"
</IfModule>

# Handle OPTIONS preflight requests
RewriteEngine On
RewriteCond %{REQUEST_METHOD} OPTIONS
RewriteRule ^(.*)$ $1 [R=200,L]

# LiteSpeed Cache - prevent caching API responses
<IfModule LiteSpeed>
    RewriteRule ^wp-json/ - [E=Cache-Control:no-cache]
</IfModule>
# END CORS for Capacitor iOS/Android App
